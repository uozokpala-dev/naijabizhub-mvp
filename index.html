function renderBusinesses(data) { businessListContainer.innerHTML = ''; if (data.length === 0) { businessListContainer.innerHTML = `

No businesses found.

`; return; } data.forEach(biz => { const card = document.createElement('div'); card.className = "p-4 bg-white border rounded-xl shadow-sm cursor-pointer"; card.innerHTML = `

${biz.name}

${biz.location}

${biz.distance}
`; card.addEventListener('click', () => openModal(biz)); businessListContainer.appendChild(card); }); }
